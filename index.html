function createModal(movie) const movieModal = document.createElement("div"); movieModal.className = "movie-modal"; movieModal.innerHTML = `

${movie.title}

${movie.overview}

개봉일 : ${movie.release_date}

평점 : ${movie.vote_average}/10

`; const modalContainer = document.getElementById("modal-content"); if (modalContainer) { modalContainer.innerHTML = ""; modalContainer.appendChild(movieModal); modalContainer.style.display = "block"; console.log("모달이 modal-content에 추가됨"); } else { console.error("modal-content 요소를 찾을 수 없습니다."); } movieModal.style.display = "block";
